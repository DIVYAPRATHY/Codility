class Solution {
    public int solution(int X, int Y, int D) {
        // write your code in Java SE 8
        if (X>=Y){
            return 0;
        }
        int remainder = (Y-X)%D;//Find the remainder 
        int step = (Y-remainder-X)/D;//Find the quotient
        if (remainder>0){
          return step+1;    
        }else {
          return step;    
        }
        
        
    }
}
